\item Excluding such confounds makes these results uninterpretable, and risks misattributions of causality {\tiny (Rowe \& Rodgers, 1997; Rutter, 2007).

\begin{frame}[fragile]{Genetic and Environmental Confounds}
\begin{itemize}\item Indeed, intelligence and AFI have sizable genetic and shared environmental variances {\tiny(Harden \& Mendle, 2011)}.\smallskip
\item Using a twin sample, Harden \& Mendle (2011) found that the relationship between intelligence and AFI was fully mediated through shared environmental differences.
\end{itemize}
\end{frame}
\section{Hypotheses}\subsection{ }
\begin{frame}[fragile]{Hypotheses}
\begin{itemize}
\item We wanted to know whether the Harden \& Mendle (2011) results would hold using a different design and data set.\smallskip
\item We examined the relationship between AFI and Intelligence using siblings and their children from a multigenerational nationally representative sample.\smallskip
\item Specifically, we addressed the following questions:\smallskip
\begin{itemize}\item Does intelligence (mother or child) predict child AFI, after controlling for gene and environmental confounds?\smallskip
\item Is this relationship consistent between-- and within--families?
\end{itemize}
\end{itemize}
\end{frame}

\subsection{Design}
\begin{frame}[fragile]{Design} 
\begin{itemize}\item We (2014) adapted Kenny's (2001) reciprocal standard dyad model.\\
\begin{align*}
&&\widehat{Y}_{diff} &= \beta_{0} + \beta_{1}\bar{Y} + \beta_{2}\bar{X} + \beta_{3}X_{diff}\\[2.5mm] 
 \text{where,}&&Y_{diff} &= Y_{1i}-Y_{2i} \text{;  } \bar{Y} = \frac{Y_{1i}+Y_{2i}}{2} \\[2.5mm] 
&&X_{diff} &= X_{1i}-X_{2i} \text{; } \bar{X} = \frac{X_{1i}+X_{2i}}{2} 
\end{align*}
\end{itemize}
\end{frame}
\begin{frame}[fragile]{Design} 
\begin{itemize}\item The relative difference in outcomes ($\widehat{Y}_{diff}$) is predicted from\smallskip
\begin{itemize}\item the mean level of the outcome ($\bar{Y}$),\smallskip 
\item the mean level of the predictor ($\bar{X}$), and \smallskip\item the between-kin predictor difference ($X_{diff}$).\end{itemize}\smallskip
\item The mean levels support causal inference through at least partial control for genes and shared environment. \smallskip
\item Therefore, we simultaneously evaluate the individual difference ($X_{diff}$) and the joint contribution of genes and shared environment ($\bar{Y}$ \& $\bar{X}$).\end{itemize}
\end{frame}

\section{Summary}
\subsection{Findings}
\begin{frame}[fragile]{Findings}
\begin{itemize}
\item Between families, \smallskip
\begin{itemize}\item both child and maternal intelligence explained child AFI.\smallskip
\item When modeled jointly, only maternal intelligence explained child AFI.\end{itemize}\smallskip
\item Within families,\smallskip
\begin{itemize}\item differences in maternal intelligence did not explain differences child AFI.\smallskip
\item differences in child intelligence explained small differences in daughter AFI ($\triangle R^{2}$=.019).\smallskip
\item When modeled jointly, differences in child intelligence \textit{still} explained small differences in daughter AFI, not son AFI.
\begin{itemize}
\item However, this difference can also be explained by non--shared enviromental factors that correlate with child intelligence.\end{itemize}
\end{itemize}
\end{itemize}
\end{frame}
